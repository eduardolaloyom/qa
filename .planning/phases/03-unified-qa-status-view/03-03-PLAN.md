  // ── Phase 3 DASH-05: Per-client trend chart ──
  let allRunsDetailed = null;  // lazy cache: array of full run objects with clients{}

  async function loadAllRunDetails() {
      if (allRunsDetailed) return allRunsDetailed;
      if (!Array.isArray(allRuns) || allRuns.length === 0) return [];
      const results = await Promise.all(
          allRuns.map(r => loadRunDetails(r.date).catch(() => null))
      );
      allRunsDetailed = results.filter(Boolean);
      return allRunsDetailed;
  }

  async function populateTrendClientSelector() {
      const sel = document.getElementById('trendClientSelector');
      if (!sel) return;
      // D-04: union de clients[slug].name observados en los últimos 30 runs (allRuns)
      // Para obtener la unión, necesitamos los detalles — forzamos el primer lazy load aquí
      // porque el usuario necesita ver el dropdown populado desde el primer render.
      const detailed = await loadAllRunDetails();
      const clientMap = {};  // { slug: name }
      for (const run of detailed) {
          if (!run || !run.clients) continue;
          for (const [slug, c] of Object.entries(run.clients)) {
              if (!clientMap[slug]) clientMap[slug] = c.name || slug;
          }
      }
      const options = [
          '<option value="" selected>Todos los clientes (agregado)</option>',
          ...Object.entries(clientMap)
              .sort(([, a], [, b]) => a.localeCompare(b))  // D-04: alfabético por name
              .map(([slug, name]) => `<option value="${escapeHtml(slug)}">${escapeHtml(name)}</option>`)
      ];
      sel.innerHTML = options.join('');
  }

  async function updateTrendChart(filterSlug = null) {
      const canvas = document.getElementById('trendChart');
      // Pitfall 1: destroy existing chart instance before creating a new one (Chart.js v4 API)
      const existing = Chart.getChart(canvas);
      if (existing) existing.destroy();

      const labels = allRuns.map(r => r.date).reverse();
      let passedData, failedData, title, passedColor;

      if (!filterSlug) {
          // AGGREGATE PATH — must be pixel-identical to pre-Phase 3 behavior
          passedData = allRuns.map(r => r.passed).reverse();
          failedData = allRuns.map(r => r.failed).reverse();
          title = `${labels.length} run${labels.length !== 1 ? 's' : ''} · últimos 30 días`;
          passedColor = '#10b981';
      } else {
          // PER-CLIENT PATH — lazy-load allRunsDetailed, map to per-client series
          const detailed = await loadAllRunDetails();
          // Align detailed array to allRuns order (both sorted desc by date from loadHistoryIndex)
          // Fallback to 0 when a run has no entry for this slug (D-05: nullish coalesce)
          const detailedByDate = {};
          for (const run of detailed) {
              if (run && run.date) detailedByDate[run.date] = run;
          }
          passedData = allRuns.map(r => detailedByDate[r.date]?.clients?.[filterSlug]?.passed ?? 0).reverse();
          failedData = allRuns.map(r => detailedByDate[r.date]?.clients?.[filterSlug]?.failed ?? 0).reverse();
          // Derive client name from any run that has it
          let clientName = filterSlug;
          for (const run of detailed) {
              if (run?.clients?.[filterSlug]?.name) { clientName = run.clients[filterSlug].name; break; }
          }
          const hasAnyData = passedData.some(v => v > 0) || failedData.some(v => v > 0);
          title = hasAnyData
              ? `${clientName} · ${labels.length} runs · últimos 30 días`
              : `Sin datos para ${clientName}`;
          passedColor = '#667eea';  // accent when filtered (UI-SPEC Dimension 3)

          type: 'line',

              labels,
